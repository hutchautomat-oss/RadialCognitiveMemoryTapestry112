### 2. The 1-Draw-Call Visual Cortex (`src/SaccadeInstancedMesh.jsx`)
This WebGL renderer completely bypasses React's virtual-DOM diffing. It directly mutates the VRAM matrix buffer inside the `useFrame` loop. It also includes the crucial `computeBoundingSphere()` patch we identified during the threat audit to prevent the "Ghost Node" raycasting bug.

```javascript
// Save exactly as: src/SaccadeInstancedMesh.jsx
import React, { useRef, useEffect } from 'react';
import { useFrame } from '@react-three/fiber';
import * as THREE from 'three';
import { useSaccadeStore } from './store';

// CTO MANDATE: Zero-allocation math utilities cached outside the runtime thread loop to prevent GC spikes
const tempObject = new THREE.Object3D();
const tempColor = new THREE.Color();
const MAX_NODES = 8000;
const STRIDE = 7; 
const hiddenMatrix = new THREE.Matrix4().makeScale(0, 0, 0); 

// Temporal Decay Invariants
const DEATH_THRESHOLD = 0.2; 

export default function SaccadeInstancedMesh() {
    const meshRef = useRef(null);
    const nodeTimestamps = useRef(new Float32Array(MAX_NODES).fill(-1.0));

    useEffect(() => {
        if (!meshRef.current) return;
    }, []);

    useFrame(() => {
        const mesh = meshRef.current;
        if (!mesh) return;

        const { activeFrameIndex, mockFrames, setVacantSlotRegistry } = useSaccadeStore.getState();
        const currentFrameData = mockFrames[activeFrameIndex];
        if (!currentFrameData) return;

        const newlyPrunedIndices = [];

        // Mutate the raw WebGL buffer directly for all 8,000 nodes
        for (let i = 0; i < MAX_NODES; i++) {
            const offset = i * STRIDE;
            let scale = currentFrameData[offset + 6];

            // Phase 3 Real-Time Memory Pruning (Exponential Decay)
            if (scale > 0 && nodeTimestamps.current[i] !== -1.0) {
                 if (scale < DEATH_THRESHOLD) {
                     scale = 0.0;
                     nodeTimestamps.current[i] = -1.0;
                     newlyPrunedIndices.push(i);
                 }
            }

            // 1. Update Spatial Matrix
            if (scale > 0) {
                tempObject.position.set(
                    currentFrameData[offset],
                    currentFrameData[offset + 1],
                    currentFrameData[offset + 2]
                );
                tempObject.scale.set(scale, scale, scale);
                tempObject.updateMatrix();
                mesh.setMatrixAt(i, tempObject.matrix);

                // 2. Update Color Matrix
                tempColor.setRGB(
                    currentFrameData[offset + 3],
                    currentFrameData[offset + 4],
                    currentFrameData[offset + 5]
                );
                mesh.setColorAt(i, tempColor);
            } else {
                // 3. Hide dead nodes
                mesh.setMatrixAt(i, hiddenMatrix);
            }
        }

        if (newlyPrunedIndices.length > 0) {
            setVacantSlotRegistry(newlyPrunedIndices);
        }

        // Flag the WebGL buffers for rendering updates
        mesh.instanceMatrix.needsUpdate = true;
        if (mesh.instanceColor) mesh.instanceColor.needsUpdate = true;
        
        // Critical Patch: Update the BVH so Raycaster doesn't click "Ghost Nodes"
        mesh.computeBoundingSphere();
    });

    return (
        <instancedMesh ref={meshRef} args={[null, null, MAX_NODES]}>
            <sphereGeometry args={[0.15, 16, 16]} />
            <meshBasicMaterial vertexColors toneMapped={false} />
        </instancedMesh>
    );
}
```
