### Step 2. The Proxy Geometry Indexer (`store.ts`)
Inject the BVH state, the VRAM-aware selection set, and the `blastSelectedSlots` action into your Zustand store. The `rebuildBVH` function creates an invisible parallel geometry where 1 Triangle = 1 Node, mapping the `triangleIndex` perfectly to the VRAM `instanceId`.

```typescript
import { create } from 'zustand';
import { MeshBVH } from 'three-mesh-bvh';
import * as THREE from 'three';

const MAX_NODES = 8000;
const STRIDE = 7; // [x, y, z, r, g, b, scale]

export const useSaccadeStore = create((set, get) => ({
    activeFrameIndex: 0,
    mockFrames: [],
    vacantSlots: [], 
    spawnTime: new Float32Array(MAX_NODES).fill(0),
    
    // NEW: BVH and Selection State
    collisionBVH: null,
    selectedSlots: new Set<number>(),

    setSelectedSlots: (slots: Set<number>) => set({ selectedSlots: slots }),

    rebuildBVH: () => {
        const { activeFrameIndex, mockFrames } = get();
        const currentFrameData = mockFrames[activeFrameIndex];
        if (!currentFrameData) return;

        // 1 Triangle = 3 Vertices = 9 Floats per instance
        const positions = new Float32Array(MAX_NODES * 9);

        for (let i = 0; i < MAX_NODES; i++) {
            const offset = i * STRIDE;
            const scale = currentFrameData[offset + 6];

            // Ghost cleanup: skip dead nodes by dumping their BVH proxy at Infinity
            if (scale <= 0.01) {
                positions.fill(Infinity, i * 9, i * 9 + 9);
                continue;
            }

            const x = currentFrameData[offset];
            const y = currentFrameData[offset + 1];
            const z = currentFrameData[offset + 2];
            const r = 0.15 * scale; // Safe bounding radius

            // Create a diagonal Proxy Triangle perfectly wrapping the AABB volume
            positions[i * 9]     = x - r; positions[i * 9 + 1] = y - r; positions[i * 9 + 2] = z - r;
            positions[i * 9 + 3] = x + r; positions[i * 9 + 4] = y + r; positions[i * 9 + 5] = z + r;
            positions[i * 9 + 6] = x - r; positions[i * 9 + 7] = y + r; positions[i * 9 + 8] = z - r;
        }

        const geometry = new THREE.BufferGeometry();
        geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));

        // maxLeafTris: 1 mathematically guarantees triangleIndex === Instance ID
        const bvh = new MeshBVH(geometry, { maxLeafTris: 1 });
        set({ collisionBVH: bvh });
    },

    blastSelectedSlots: () => {
        const { activeFrameIndex, mockFrames, selectedSlots, vacantSlots } = get();
        const currentFrameBuffer = mockFrames[activeFrameIndex];
        if (!currentFrameBuffer || selectedSlots.size === 0) return;

        const reclaimed = [];

        selectedSlots.forEach((index) => {
            const offset = index * STRIDE;
            // 1. Force visual scale to 0.0 (VRAM purge)
            currentFrameBuffer[offset + 6] = 0.0;
            // 2. Queue for FIFO reclamation
            reclaimed.push(index);
        });

        set((state) => ({
            // Append newly freed indices into the vacancy array pool
            vacantSlots: Array.from(new Set([...state.vacantSlots, ...reclaimed])),
            selectedSlots: new Set() // Clear selection after blast
        }));

        // Rebuild BVH to instantly remove blasted nodes from spatial index
        get().rebuildBVH();
    },

    // ... [existing injectLiveIntentVector logic]
}));
```

### Step 3. The 8k BVH Lasso Consumer (`LassoSelection.tsx`)
Because we migrated entirely to the `InstancedMesh`, we completely bypass React filtering. Hook this logic into your screen-polygon drawing mechanics to traverse the tree in $O(\log N)$ time.

```typescript
import * as THREE from 'three';
import { useSaccadeStore } from './store';

export function executeLassoHitTest(frustum: THREE.Frustum): void {
    const { collisionBVH, setSelectedSlots } = useSaccadeStore.getState();
    const hits = new Set<number>();
    
    if (!collisionBVH) return;

    collisionBVH.shapecast({
        // Broad-phase: Instantly discard BVH branches entirely outside the lasso
        intersectsBounds: (box) => frustum.intersectsBox(box),
        
        // Narrow-phase: Collect the exact Node ID
        intersectsTriangle: (tri, index) => {
            hits.add(index);  // Index precisely matches the VRAM Instance ID!
            return false;     // Return false to continue evaluating other branches
        }
    });

    setSelectedSlots(hits);
}
```
